# SSH конфигурация для проекта Sculptura
# Использование: ssh sculptura
# 
# ВАЖНО: Замените примеры на реальные значения:
# - HostName: IP адрес или домен вашего сервера
# - User: имя пользователя для SSH подключения
# - IdentityFile: путь к вашему приватному SSH ключу

Host sculptura
    HostName your-server.com
    Port 22
    User your-username
    IdentityFile ~/.ssh/id_rsa
    IdentitiesOnly yes
